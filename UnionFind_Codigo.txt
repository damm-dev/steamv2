/**
 * Estructura de Datos: Union-Find (Disjoint Set Union - DSU)
 * 
 * Implementación en Java utilizando genéricos para admitir cualquier tipo de datos.
 * Incluye optimizaciones de Compresión de Caminos (Path Compression) y Unión por Rango (Union by Rank).
 */

import java.util.HashMap;
import java.util.Map;

public class UnionFind<T> {

    // Mapea cada elemento a su representante (padre)
    private final Map<T, T> parent = new HashMap<>();
    
    // Mapea cada representante a su rango (altura aproximada del árbol)
    private final Map<T, Integer> rank = new HashMap<>();

    /**
     * Crea un nuevo conjunto disjunto para el elemento especificado si no existe previamente.
     */
    public void makeSet(T element) {
        if (!parent.containsKey(element)) {
            parent.put(element, element);
            rank.put(element, 0);
        }
    }

    /**
     * Encuentra y retorna el representante del conjunto al que pertenece el elemento.
     * Aplica Compresión de Caminos para optimizar futuras búsquedas.
     */
    public T find(T element) {
        if (!parent.containsKey(element)) {
            return null;
        }

        T p = parent.get(element);
        if (p.equals(element)) {
            return element;
        }
   
        // Compresión de caminos: asignar la raíz directamente como padre
        T root = find(p);
        parent.put(element, root);
        return root;
    }

    /**
     * Une los conjuntos de los dos elementos. Retorna true si se realizó la unión, 
     * o false si ya pertenecían al mismo conjunto o si alguno no existe.
     * Aplica Unión por Rango para mantener los árboles balanceados.
     */
    public boolean union(T element1, T element2) {
        T root1 = find(element1);
        T root2 = find(element2);

        if (root1 == null || root2 == null || root1.equals(root2)) {
            return false;
        }

        int rank1 = rank.get(root1);
        int rank2 = rank.get(root2);

        // Unión por rango (el árbol de menor rango se cuelga del de mayor rango)
        if (rank1 < rank2) {
            parent.put(root1, root2);
        } else if (rank1 > rank2) {
            parent.put(root2, root1);
        } else {
            parent.put(root2, root1);
            rank.put(root1, rank1 + 1);
        }

        return true;
    }

    /**
     * Comprueba si dos elementos pertenecen al mismo conjunto (están conectados).
     */
    public boolean connected(T element1, T element2) {
        T root1 = find(element1);
        T root2 = find(element2);
        return root1 != null && root1.equals(root2);
    }
}
